        🔗 External Integration

        Link a Google Sheet (via Apps Script) or any webhook URL. Every attendance event will be pushed automatically.

        📋 Google Sheets setup guide
        1. Open your Google Sheet → Extensions → Apps Script
        2. Paste the Apps Script below and deploy as a Web App (Anyone can access):
        function doPost(e) {
  var data = JSON.parse(e.postData.contents);
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(['Date','Staff ID','Name','Action','Time','Lat','Lng']);
  }
  var ts = new Date(data.timestamp);
  sheet.appendRow([
    ts.toLocaleDateString(), data.staff_id, data.name || '',
    data.action, ts.toLocaleTimeString(),
    data.lat || '', data.lng || ''
  ]);
  return ContentService.createTextOutput('ok');
}
